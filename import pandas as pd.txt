import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import classification_report, confusion_matrix, accuracy_score

# 1. Load your dataset
# Ensure you have the 'creditcard.csv' file in your directory
df = pd.read_csv('creditcard.csv')

# 2. Preprocessing: Scale the 'Amount' and 'Time' features
scaler = StandardScaler()
df['NormalizedAmount'] = scaler.fit_transform(df['Amount'].values.reshape(-1, 1))
df.drop(['Time', 'Amount'], axis=1, inplace=True) # Remove original unscaled columns

# 3. Define Features (X) and Target (y)
X = df.drop('Class', axis=1)
y = df['Class']

# 4. Split data into Training and Testing sets (70% train, 30% test)
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# 5. Initialize and Train the Random Forest Model
# Use 'class_weight="balanced"' to help handle the data imbalance
rfc = RandomForestClassifier(n_estimators=100, random_state=42, class_weight='balanced')
rfc.fit(X_train, y_train)

# 6. Make Predictions
y_pred = rfc.predict(X_test)

# 7. Evaluate Model Performance
print("Accuracy Score:", accuracy_score(y_test, y_pred))
print("\nConfusion Matrix:\n", confusion_matrix(y_test, y_pred))
print("\nClassification Report:\n", classification_report(y_test, y_pred))
